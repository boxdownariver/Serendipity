122NCURSES_EXPORT(int)
123_nc_access(const char *path, int mode)
124{
125 int result;
126
127 if (path == 0) {
128 result = -1;
129 } else if (ACCESS(path, mode) < 0) {
130 if ((mode & W_OK) != 0
131 && errno == ENOENT
132 && strlen(path) < PATH_MAX) {
133 char head[PATH_MAX];
134 char *leaf;
135
136 _nc_STRCPY(head, path, sizeof(head));
137 leaf = _nc_basename(head);
138 if (leaf == 0)
139 leaf = head;
140 *leaf = '\0';
141 if (head == leaf)
142 _nc_STRCPY(head, ".", sizeof(head));
143
144 result = ACCESS(head, R_OK | W_OK | X_OK);
145 } else {
146 result = -1;
147 }
148 } else {
149 result = 0;
150 }
151 return result;
152}
