158 // Duplicate the field at a new position
159 inline NCursesFormField* dup(int first_row, int first_col)
160 {
161 NCursesFormField* f = new NCursesFormField();
162 if (!f)
163 OnError(E_SYSTEM_ERROR);
164 else {
