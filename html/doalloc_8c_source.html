46_nc_doalloc(void *oldp, size_t amount)
47{
48 void *newp;
49
50 if (oldp != NULL) {
51 if (amount == 0) {
52 free(oldp);
53 newp = NULL;
54 } else if ((newp = realloc(oldp, amount)) == 0) {
55 free(oldp);

58 } else {
