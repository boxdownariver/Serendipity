54{
55 FIELDTYPE *nftyp = (FIELDTYPE *)0;

57 T((T_CALLED("link_fieldtype(%p,%p)"), (void *)type1, (void *)type2));

62 if (nftyp)

65 *nftyp = *_nc_Default_FieldType;
66 SetStatus(nftyp, _LINKED_TYPE);
