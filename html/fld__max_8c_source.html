38/*---------------------------------------------------------------------------
39| Facility : libnform
40| Function : int set_max_field(FIELD *field, int maxgrow)
41|
42| Description : Set the maximum growth for a dynamic field. If maxgrow=0
43| the field may grow to any possible size.
44|
45| Return Values : E_OK - success
46| E_BAD_ARGUMENT - invalid argument
47+--------------------------------------------------------------------------*/
48FORM_EXPORT(int)
49set_max_field(FIELD *field, int maxgrow)
50{
51 T((T_CALLED("set_max_field(%p,%d)"), (void *)field, maxgrow));
52
53 if (!field || (maxgrow < 0))
54 RETURN(E_BAD_ARGUMENT);
55 else
56 {
57 bool single_line_field = Single_Line_Field(field);
58
59 if (maxgrow > 0)
60 {
61 if (((single_line_field && (maxgrow < field->dcols)) ||
62 (!single_line_field && (maxgrow < field->drows))) &&
63 !Field_Has_Option(field, O_INPUT_LIMIT))
64 RETURN(E_BAD_ARGUMENT);
65 }
66 field->maxgrow = maxgrow;
67 /* shrink */
68 if (maxgrow > 0 && Field_Has_Option(field, O_INPUT_LIMIT) &&
69 field->dcols > maxgrow)
70 field->dcols = maxgrow;
71 ClrStatus(field, _MAY_GROW);
72 if (!((unsigned)field->opts & O_STATIC))
73 {
74 if ((maxgrow == 0) ||
75 (single_line_field && (field->dcols < maxgrow)) ||
76 (!single_line_field && (field->drows < maxgrow)))
77 SetStatus(field, _MAY_GROW);
78 }
79 }
80 RETURN(E_OK);
81}
