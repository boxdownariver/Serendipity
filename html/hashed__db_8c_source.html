63static DB *
64find_connection(const char *path, bool modify)
65{
66 DB *result = 0;
67 MYCONN *p;
68
69 for (p = connections; p != 0; p = p->next) {
70 if (!strcmp(p->path, path) && p->modify == modify) {
71 result = p->db;
72 break;
73 }
74 }
75
76 return result;
77}

79static void
80drop_connection(DB * db)
81{
82 MYCONN *p, *q;
83
84 for (p = connections, q = 0; p != 0; q = p, p = p->next) {
85 if (p->db == db) {
86 if (q != 0)
87 q->next = p->next;
88 else
89 connections = p->next;
90 free(p->path);
91 free(p);
92 break;
93 }
94 }
95}

97static void
98make_connection(DB * db, const char *path, bool modify)
99{
100 MYCONN *p = typeCalloc(MYCONN, 1);
101
102 if (p != 0) {
103 p->db = db;
104 p->path = strdup(path);
105 p->modify = modify;
106 if (p->path != 0) {
107 p->next = connections;
108 connections = p;
109 } else {
110 free(p);
111 }
112 }
113}

140#elif HASHED_DB_API >= 3

148 result = 0;
149 }
150#elif HASHED_DB_API >= 2
151 if ((code = db_open(path,
152 DB_HASH,
153 modify ? DB_CREATE : DB_RDONLY,
154 0644,

156 (DB_INFO *) 0,
157 &result)) != 0) {
158 result = 0;
159 }
160#else
161 if ((result = dbopen(path,
162 modify ? (O_CREAT | O_RDWR) : O_RDONLY,
163 0644,
164 DB_HASH,

166 code = errno;
167 }
168#endif
169 if (result != 0) {
170 make_connection(result, path, modify);
171 T(("opened %s", path));
172 } else {
173 T(("cannot open %s: %s", path, strerror(code)));
174 }
175 }
