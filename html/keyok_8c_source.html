39/*
40 * Enable (or disable) ncurses' interpretation of a keycode by adding (or
41 * removing) the corresponding 'tries' entry.
42 *
43 * Do this by storing a second tree of tries, which records the disabled keys.
44 * The simplest way to copy is to make a function that returns the string (with
45 * nulls set to 0200), then use that to reinsert the string into the
46 * corresponding tree.
47 */
48

50NCURSES_SP_NAME(keyok) (NCURSES_SP_DCLx int c, bool flag)
51{
52 int code = ERR;
53
54 if (HasTerminal(SP_PARM)) {
55 T((T_CALLED("keyok(%p, %d,%d)"), (void *) SP_PARM, c, flag));
56#ifdef USE_TERM_DRIVER
57 code = CallDriver_2(sp, td_kyOk, c, flag);
58#else
59 if (c >= 0) {
60 int count = 0;
61 char *s;
62 unsigned ch = (unsigned) c;
63
64 if (flag) {
65 while ((s = _nc_expand_try(SP_PARM->_key_ok,
66 ch, &count, (size_t) 0)) != 0) {
67 if (_nc_remove_key(&(SP_PARM->_key_ok), ch)) {
68 code = _nc_add_to_try(&(SP_PARM->_keytry), s, ch);
69 free(s);
70 count = 0;
71 if (code != OK)
72 break;
73 } else {
74 free(s);
75 }
76 }
77 } else {
78 while ((s = _nc_expand_try(SP_PARM->_keytry,
79 ch, &count, (size_t) 0)) != 0) {
80 if (_nc_remove_key(&(SP_PARM->_keytry), ch)) {
81 code = _nc_add_to_try(&(SP_PARM->_key_ok), s, ch);

83 count = 0;

92#endif

95}
