55 * Sound the current terminal's audible bell if it has one. If not,
56 * flash the screen if possible.
57 *
58 */
59
60NCURSES_EXPORT(int)
61NCURSES_SP_NAME(beep) (NCURSES_SP_DCL0)
62{
63 int res = ERR;
64
65 T((T_CALLED("beep(%p)"), (void *) SP_PARM));
66
67#ifdef USE_TERM_DRIVER
68 if (SP_PARM != 0)
69 res = CallDriver_1(SP_PARM, td_doBeepOrFlash, TRUE);
70#else
71 /* FIXME: should make sure that we are not in altchar mode */
72 if (cur_term == 0) {
73 res = ERR;
74 } else if (bell) {

80#endif

82 returnCode(res);
83}
