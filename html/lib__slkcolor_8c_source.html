44static int
45_nc_slk_color(SCREEN *sp, int pair_arg)
46{
47 int code = ERR;
48
49 T((T_CALLED("slk_color(%p,%d)"), (void *) sp, pair_arg));
50
51 if (sp != 0
52 && sp->_slk != 0
53 && pair_arg >= 0
54 && pair_arg < sp->_pair_limit) {
55 TR(TRACE_ATTRS, ("... current is %s", _tracech_t(CHREF(sp->_slk->attr))));
56 SetPair(sp->_slk->attr, pair_arg);
57 TR(TRACE_ATTRS, ("new attribute is %s", _tracech_t(CHREF(sp->_slk->attr))));
58 code = OK;
59 }
60 returnCode(code);
61}
