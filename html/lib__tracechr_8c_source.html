50_nc_tracechar(SCREEN *sp, int ch)
51{
52 NCURSES_CONST char *name;
53 char *MyBuffer = ((sp != 0)
54 ? sp->tracechr_buf
55 : _nc_globals.tracechr_buf);
56
57 if (ch > KEY_MIN || ch < 0) {
58 name = safe_keyname(SP_PARM, ch);
59 if (name == 0 || *name == '\0')
60 name = "NULL";
61 _nc_SPRINTF(MyBuffer, _nc_SLIMIT(MyBufSize)
62 "'%.30s' = %#03o", name, ch);
63 } else if (!is8bits(ch) || !isprint(UChar(ch))) {
64 /*
65 * workaround for glibc bug:
66 * sprintf changes the result from unctrl() to an empty string if it
67 * does not correspond to a valid multibyte sequence.
68 */
69 _nc_SPRINTF(MyBuffer, _nc_SLIMIT(MyBufSize)
70 "%#03o", ch);
71 } else {
72 name = safe_unctrl(SP_PARM, (chtype) ch);
73 if (name == 0 || *name == 0)
74 name = "null"; /* shouldn't happen */
75 _nc_SPRINTF(MyBuffer, _nc_SLIMIT(MyBufSize)
76 "'%.30s' = %#03o", name, ch);
77 }
78 return (MyBuffer);
79}
