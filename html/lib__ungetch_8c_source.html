64 int rc = ERR;

66 T((T_CALLED("ungetch(%p,%s)"), (void *) sp, _nc_tracechar(sp, ch)));
67
68 if (sp != 0 && tail >= 0) {

71 t_inc();

74 h_dec();

76
77 sp->_fifo[head] = ch;
78 T(("ungetch %s ok", _nc_tracechar(sp, ch)));
