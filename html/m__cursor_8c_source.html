52+--------------------------------------------------------------------------*/
53MENU_EXPORT(int)
54_nc_menu_cursor_pos(const MENU *menu, const ITEM *item, int *pY, int *pX)
55{
56 if (!menu || !pX || !pY)
57 return (E_BAD_ARGUMENT);
58 else
59 {
60 if ((ITEM *)0 == item)
61 item = menu->curitem;
62 assert(item != (ITEM *)0);
63
64 if (!(menu->status & _POSTED))
65 return (E_NOT_POSTED);
66
67 *pX = item->x * (menu->spc_cols + menu->itemlen);
68 *pY = (item->y - menu->toprow) * menu->spc_rows;
69 }
70 return (E_OK);
71}
