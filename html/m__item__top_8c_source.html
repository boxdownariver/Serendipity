43/*---------------------------------------------------------------------------
44| Facility : libnmenu
45| Function : int set_top_row(MENU *menu, int row)
46|
47| Description : Makes the specified row the top row in the menu
48|
49| Return Values : E_OK - success
50| E_BAD_ARGUMENT - not a menu pointer or invalid row
51| E_NOT_CONNECTED - there are no items for the menu
52+--------------------------------------------------------------------------*/
53MENU_EXPORT(int)
54set_top_row(MENU *menu, int row)
55{
56 T((T_CALLED("set_top_row(%p,%d)"), (void *)menu, row));
57
58 if (menu)
59 {
60 if (menu->status & _IN_DRIVER)
61 RETURN(E_BAD_STATE);
62 if (menu->items == (ITEM **)0)
63 RETURN(E_NOT_CONNECTED);
64
65 if ((row < 0) || (row > (menu->rows - menu->arows)))
66 RETURN(E_BAD_ARGUMENT);
67 }
68 else
69 RETURN(E_BAD_ARGUMENT);
70
71 if (row != menu->toprow)
72 {
73 ITEM *item;
74
75 if (menu->status & _LINK_NEEDED)
76 _nc_Link_Items(menu);
77
78 item = menu->items[(menu->opt & O_ROWMAJOR) ? (row * menu->cols) : row];
79 assert(menu->pattern);
80 Reset_Pattern(menu);
81 _nc_New_TopRow_and_CurrentItem(menu, row, item);
82 }
83
84 RETURN(E_OK);
85}

95MENU_EXPORT(int)
96top_row(const MENU *menu)
97{
98 T((T_CALLED("top_row(%p)"), (const void *)menu));
99 if (menu && menu->items && *(menu->items))
100 {
101 assert((menu->toprow >= 0) && (menu->toprow < menu->rows));

105 returnCode(ERR);
