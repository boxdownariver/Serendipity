43/*---------------------------------------------------------------------------
44| Facility : libnmenu
45| Function : int set_item_value(ITEM *item, int value)
46|
47| Description : Programmatically set the item's selection value. This is
48| only allowed if the item is selectable at all and if
49| it is not connected to a single-valued menu.
50| If the item is connected to a posted menu, the menu
51| will be redisplayed.
52|
53| Return Values : E_OK - success
54| E_REQUEST_DENIED - not selectable or single valued menu
55+--------------------------------------------------------------------------*/
56MENU_EXPORT(int)
57set_item_value(ITEM *item, bool value)
58{
59 T((T_CALLED("set_item_value(%p,%d)"), (void *)item, value));
60 if (item)
61 {
62 MENU *menu = item->imenu;
63
64 if ((!(item->opt & O_SELECTABLE)) ||
65 (menu && (menu->opt & O_ONEVALUE)))
66 RETURN(E_REQUEST_DENIED);
67
68 if (item->value ^ value)
69 {
70 item->value = value ? TRUE : FALSE;
71 if (menu)
72 {
73 if (menu->status & _POSTED)
74 {
75 Move_And_Post_Item(menu, item);
76 _nc_Show_Menu(menu);
77 }
78 }
79 }
80 }
81 else
82 _nc_Default_Item.value = value;
83
84 RETURN(E_OK);
85}
