171void createMenu(MENU *&mainMenu, WINDOW *mainWindow,
172 const MenuLines &mainMenuInfo, ITEM **&items) {
173 size_t menuLineSize, i; //INPUT- Menu size, iterator
174 int cols; //OUTPUT- Columns of current window
175
176 //Init cols, menuLineSize & items
177 cols = getmaxx(mainWindow);
178 menuLineSize = mainMenuInfo.menuLines.size();
179 items = (ITEM**) calloc(menuLineSize + 1, sizeof(ITEM*));
180
181 //Items must be initialized to menu line items
182 for (i = 0; i < menuLineSize; i++) {
183 items[i] = new_item(mainMenuInfo.menuLines[i].c_str(),
184 mainMenuInfo.menuLines[i].c_str());
185 }
186 items[menuLineSize] = new_item((char*) NULL, (char*) NULL);
187
188 //Initialize menu, apply settings
189 mainMenu = new_menu((ITEM**) items);
190 set_menu_win(mainMenu, mainWindow);
191 set_menu_sub(mainMenu,
192 derwin(mainWindow, menuLineSize,
193 mainMenuInfo.longestMenuLength, 4,
194 (cols - mainMenuInfo.longestMenuLength) / 2));
195 set_menu_mark(mainMenu, ">");
196 post_menu(mainMenu);
197}
