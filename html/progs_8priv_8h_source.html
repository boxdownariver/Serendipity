242#define use_verbosity(level) do { \
243 set_trace_level(level); \
244 if (_nc_tracing > DEBUG_LEVEL(2)) \
245 _nc_tracing |= TRACE_MAXIMUM; \
246 else if (_nc_tracing == DEBUG_LEVEL(2)) \
247 _nc_tracing |= TRACE_ORDINARY; \
248 if (level >= 2) \

250 } while (0)
