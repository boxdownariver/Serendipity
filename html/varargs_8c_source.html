72 if (MyBuffer == 0)
73 return dummy;

76 while (*fmt != '\0') {
77 if (*fmt == '%') {
78 char *pval = 0; /* avoid const-cast */

93 parm[parms++] = atInteger;
94 } else if (isalpha(UChar(*fmt))) {
95 done = TRUE;
96 switch (*fmt) {
97 case 'Z': /* FALLTHRU */
98 case 'h': /* FALLTHRU */
99 case 'l': /* FALLTHRU */
100 done = FALSE;
101 type = *fmt;
102 break;
103 case 'i': /* FALLTHRU */
104 case 'd': /* FALLTHRU */
105 case 'u': /* FALLTHRU */
106 case 'x': /* FALLTHRU */
107 case 'X': /* FALLTHRU */

145 }

182 }

184 fmt++;
185 }
